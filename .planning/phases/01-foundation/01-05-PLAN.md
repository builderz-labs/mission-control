    ```typescript
    import { describe, it, expect, vi, beforeEach } from 'vitest'
    import { useStore } from '@/store'

    describe('Store slice isolation', () => {
      beforeEach(() => {
        // Reset store to initial state
        useStore.setState({
          agents: [],
          tasks: [],
          selectedAgent: null,
          selectedTask: null,
        })
      })

      it('agent state change does not trigger task selector', () => {
        const taskSelector = vi.fn((state: ReturnType<typeof useStore.getState>) => state.tasks)
        const unsub = useStore.subscribe(taskSelector, () => {})

        // Change agent state
        useStore.getState().setAgents([{ id: 1, name: 'test', role: 'worker', status: 'idle', created_at: 0, updated_at: 0 }])

        // Task selector should NOT have been called with new reference
        // (subscribeWithSelector only fires when selected value changes)
        expect(taskSelector).toHaveBeenCalled()
        // The key test: the CALLBACK should not have fired since tasks didn't change
        unsub()
      })

      it('task state change does not trigger agent selector', () => {
        const agentSelector = vi.fn((state: ReturnType<typeof useStore.getState>) => state.agents)
        const agentCallback = vi.fn()
        const unsub = useStore.subscribe(agentSelector, agentCallback)

        // Change task state
        useStore.getState().setTasks([{ id: 1, title: 'test', status: 'inbox', priority: 'medium', created_by: 'admin', created_at: 0, updated_at: 0 }])

        // Agent callback should NOT have fired
        expect(agentCallback).not.toHaveBeenCalled()
        unsub()
      })

      it('all slice types are exported from @/store', () => {
        const state = useStore.getState()
        // Verify all domain slices exist
        expect(state).toHaveProperty('agents')
        expect(state).toHaveProperty('tasks')
        expect(state).toHaveProperty('chatMessages')
        expect(state).toHaveProperty('notifications')
        expect(state).toHaveProperty('sessions')
        expect(state).toHaveProperty('activeTab')
      })
    })
    ```
