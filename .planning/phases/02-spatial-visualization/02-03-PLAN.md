### Task 1: Dagre layout utility
Create `src/lib/spatial-layout.ts`:

```typescript
import dagre from '@dagrejs/dagre'
import type { AgentNode, AgentEdge } from '@/store/canvas-store'

export function applyDagreLayout(
  nodes: AgentNode[],
  edges: AgentEdge[],
  options?: { direction?: 'TB' | 'LR'; nodeWidth?: number; nodeHeight?: number }
): { nodes: AgentNode[]; edges: AgentEdge[] } {
  const g = new dagre.graphlib.Graph()
  g.setDefaultEdgeLabel(() => ({}))
  g.setGraph({ rankdir: options?.direction ?? 'TB', nodesep: 80, ranksep: 100 })

  const nodeWidth = options?.nodeWidth ?? 180
  const nodeHeight = options?.nodeHeight ?? 60

  for (const node of nodes) {
    g.setNode(node.id, { width: nodeWidth, height: nodeHeight })
  }
  for (const edge of edges) {
    g.setEdge(edge.source, edge.target)
  }

  dagre.layout(g)

  const layoutNodes = nodes.map((node) => {
    const pos = g.node(node.id)
    return { ...node, position: { x: pos.x - nodeWidth / 2, y: pos.y - nodeHeight / 2 } }
  })

  return { nodes: layoutNodes, edges }
}
```

### Task 2: SSE hook for spatial events
Create `src/hooks/use-spatial-sse.ts`:

Hook that subscribes to `/api/events` SSE stream and dispatches canvas store updates when spatial events arrive:
- `agent.created` → add node
- `agent.updated` → update node data (status, name)
- `agent.deleted` → remove node
- `spatial.edge.added` → add edge
- `spatial.edge.removed` → remove edge

```typescript
'use client'
import { useEffect } from 'react'
import { useCanvasStore } from '@/store/canvas-store'

export function useSpatialSSE() {
  useEffect(() => {
    const es = new EventSource('/api/events')

    es.onmessage = (event) => {
      const data = JSON.parse(event.data)
      const store = useCanvasStore.getState()

      switch (data.type) {
        case 'agent.created':
          store.addNode({
            id: `agent-${data.data.id}`,
            type: 'agent',
            position: { x: 0, y: 0 },  // will be laid out by dagre
            data: { label: data.data.name, status: data.data.status, role: data.data.role, agentId: data.data.id },
          })
          break
        case 'agent.updated':
          store.updateNodeData(`agent-${data.data.id}`, { status: data.data.status })
          break
        case 'agent.deleted':
          store.removeNode(`agent-${data.data.id || data.data.name}`)
          break
      }
    }

    return () => es.close()
  }, [])
}
```
